EM ORDEM
	1-CRIAR IMAGEM
	2-CRIAR CONTAINER
	3-DEPOIS NO FINAL ATUALIZAR

CRIAR A IMAGEM NO DOCKER:
	docker-compose build --no-cache --force-rm

PARAR O CONTAINER DE RODAR 
	docker-compose stop

CRIAR O CONTAINER A PARTIR DA IMAGEM
	docker-compose up -d

ATUALIZAR O CONTAINER:
	docker exec laravel-electron bash -c "composer update"

PARA O BANCO DE DADOS
	docker exec laravel-electron bash -c "php artisan migrate"
	docker exec laravel-electron bash -c "php artisan db:seed"
