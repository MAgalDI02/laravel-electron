EM ORDEM
	1-CRIAR IMAGEM
	2-CRIAR CONTAINER

CRIAR A IMAGEM NO DOCKER:
	docker-compose build --no-cache --force-rm

CRIAR O CONTAINER A PARTIR DA IMAGEM
	docker-compose up -d


CRIAR A KEY DO PROJETO:
	docker exec laravel-electron bash -c "php artisan key:generate"

PARA O BANCO DE DADOS
	docker exec laravel-electron bash -c "php artisan migrate --seed"
